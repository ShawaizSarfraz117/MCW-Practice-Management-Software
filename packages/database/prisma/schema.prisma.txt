/// Core appointment entity representing scheduled sessions, events, or meetings.
/// Appointments are always associated with a ClientGroup (not individual clients) to support family/couple therapy.
model Appointment {
  id                       String             @id @default(dbgenerated("newid()"), map: "PK_Appointment_ID") @db.UniqueIdentifier
  /// Type of appointment:
  /// - 'APPOINTMENT': Client sessions (billable, clinical).
  /// - 'EVENT': Non-client activities like staff meetings, trainings, or clinician unavailability blocks. [cite: 1]
  /// Refer to schema_notes.md section 8.3 for defined values.
  type                     String             @db.VarChar(50)
  /// Display title for the appointment.
  /// For 'EVENT' type: e.g., "Staff Meeting", "Clinician Unavailability". [cite: 1]
  /// For 'APPOINTMENT' type: Used if distinct from PracticeService.name, e.g., "Initial Consultation - John D."
  /// See schema_notes.md 7.2.3.
  title                    String?            @db.VarChar(255) // Increased length for more descriptive event titles.
  /// Whether this spans the entire day (affects calendar display).
  is_all_day               Boolean            @default(false, map: "DF_Appointment_IsAllDay")
  /// Start date/time in UTC. All timestamps are stored in UTC.
  start_date               DateTime
  /// End date/time in UTC. All timestamps are stored in UTC.
  end_date                 DateTime
  /// Physical location where appointment takes place.
  location_id              String?            @db.UniqueIdentifier
  /// User (typically clinician or staff) who created this appointment record.
  created_by               String             @db.UniqueIdentifier
  /// Current status: e.g., 'SCHEDULED', 'COMPLETED', 'CANCELLED', 'NO_SHOW'.
  /// Refer to schema_notes.md section 8.3 for defined values and section 4.2.3 for lifecycle.
  status                   String             @db.VarChar(100)
  /// Clinician providing the service or primary host/subject of an 'EVENT'.
  /// For 'EVENT' type 'Clinician Unavailability', this is the clinician who is unavailable. [cite: 1]
  clinician_id             String?            @db.UniqueIdentifier
  /// Base fee for the appointment in dollars (before adjustments). Stored as Decimal(10,2). Derived from service rates, potentially overridden.
  appointment_fee          Decimal?           @db.Decimal(10,2)
  /// Type of service being provided (links to service catalog). Null for 'EVENT' type appointments.
  service_id               String?            @db.UniqueIdentifier
  /// Whether this is part of a recurring series.
  is_recurring             Boolean            @default(false, map: "DF_Appointment_IsRecurring")
  /// iCalendar RRULE format string defining recurrence pattern.
  /// Application logic parses this to generate/manage instances.
  /// For 'EVENT' type 'Clinician Unavailability', this defines recurring blocks of unavailable time. [cite: 1]
  /// Modification to recurring unavailability may require manual resolution of conflicts with existing client appointments. [cite: 1]
  /// See schema_notes.md section 4.2.2, 7.2.2 and Glossary (13).
  recurring_rule           String?            @db.Text
  /// For updates to recurring series:
  /// - true: cancel future appointments/events.
  /// - false: attempt to reschedule them based on the new rule (may require conflict resolution). [cite: 1]
  cancel_appointments      Boolean?
  /// Whether to send cancellation notifications when an appointment (especially a recurring one) is cancelled.
  notify_cancellation      Boolean?
  /// For recurring appointments: NULL if this is the parent/template, or parent's ID if this is a child instance.
  /// Manages parent-child relationship for recurring series.
  recurring_appointment_id String?            @db.UniqueIdentifier
  /// Client group (family/couple/individual) receiving services.
  /// Every 'APPOINTMENT' type is associated with a ClientGroup. Null for 'EVENT' type. [cite: 1]
  client_group_id          String?            @db.UniqueIdentifier
  /// Session-specific fee adjustment in dollars (can be negative for discounts/credits). Stored as Decimal(10,2).
  /// See Glossary (13) for 'Adjustment'.
  adjustable_amount        Decimal?           @db.Decimal(10,2)
  /// Link to superbill if this appointment is included in one.
  superbill_id             String?            @db.UniqueIdentifier
  /// Amount written off (not expected to be collected) in dollars. Stored as Decimal(10,2).
  /// See Glossary (13) for 'Write-off'.
  write_off                Decimal?           @db.Decimal(10,2)
  ClientGroup              ClientGroup?       @relation(fields: [client_group_id], references: [id], onDelete: NoAction, onUpdate: NoAction, map: "FK_Appointment_ClientGroup")
  Clinician                Clinician?         @relation(fields: [clinician_id], references: [id], onDelete: NoAction, onUpdate: NoAction, map: "FK_Appointment_Clinician")
  Location                 Location?          @relation(fields: [location_id], references: [id], map: "FK_Appointment_Location")
  PracticeService          PracticeService?   @relation(fields: [service_id], references: [id], map: "FK_Appointment_PracticeService")
  /// Self-referential relation for recurring appointments (parent-child hierarchy).
  Appointment              Appointment?       @relation("AppointmentToAppointment", fields: [recurring_appointment_id], references: [id], onDelete: NoAction, onUpdate: NoAction, map: "FK_Appointment_RecurringAppointment")
  other_Appointment        Appointment[]      @relation("AppointmentToAppointment")
  Superbill                Superbill?         @relation(fields: [superbill_id], references: [id], onDelete: NoAction, onUpdate: NoAction, map: "FK_Appointment_Superbill")
  User                     User               @relation(fields: [created_by], references: [id], map: "FK_Appointment_User")
  AppointmentNotes         AppointmentNotes[]
  AppointmentTag           AppointmentTag[]
  Invoice                  Invoice[]
  /// Link to survey answers, e.g., for pre-session questionnaires associated with this appointment.
  SurveyAnswers            SurveyAnswers[]
}

/// HIPAA compliance audit trail for tracking access and events related to protected health information (PHI).
model Audit {
  Id         String   @id(map: "PK_Audit_ID") @default(dbgenerated("newid()"), map: "DF_Audit_ID") @db.UniqueIdentifier
  /// Client whose data was accessed or affected (null for system-level events not specific to one client).
  client_id  String?  @db.UniqueIdentifier
  /// User who performed the action or triggered the event.
  user_id    String?  @db.UniqueIdentifier
  /// Timestamp of the event in UTC.
  datetime   DateTime @default(now(), map: "DF__Audit__datetime__17ED6F58") @db.DateTime
  /// Type of event (e.g., 'VIEW', 'UPDATE', 'DELETE', 'LOGIN_SUCCESS', 'PHI_ACCESS', 'DOCUMENT_SIGNED').
  /// Consider making this an Enum (see schema_notes.md 5.2).
  event_type String?  @db.NChar(10)
  /// Human-readable description of the event (e.g., "Viewed client record", "Updated appointment X").
  event_text String   @db.NVarChar(255)
  /// Whether this event involves HIPAA-protected information, requiring stricter logging.
  is_hipaa   Boolean  @default(false, map: "DF_Audit_IsHipaa")
  Client     Client?  @relation(fields: [client_id], references: [id], onDelete: NoAction, onUpdate: NoAction, map: "FK_Audit_Client")
  User       User?    @relation(fields: [user_id], references: [id], onDelete: NoAction, onUpdate: NoAction, map: "FK_Audit_User")
}

  created_at                 DateTime    @default(now(), map: "DF__GroupClie__creat__6EEB59C5")
